  ## Ansible - Install MariaDB

  ---
  - name: Install MariaDB for Rushi
    hosts: all
    become: yes

    tasks:
      - name: Install MariaDB
        yum:
          name: mariadb-server
          state: present

      - name: Start MariaDB
        service:
          name: mariadb
          state: started
          enabled: yes

# Notifications

## Jenkins Pipeline - Email Notifications
You will learn how to **send email notification on failure or success**.

pipeline {
  agent any
  environment {
    INSTITUTE_NAME = "Rushi"
  }
  stages {
    stage('Checkout') {
      steps {
        git url: 'https://github.com/rushikeshsatwadhar/Login-Signup-Form'
      }
    }
    stage('Build') {
      steps {
        sh 'mvn clean package'
      }
    }
  }
  post {
    success {
      mail to: 'team@rushi.com',
          subject: "SUCCESS: Build #${env.BUILD_NUMBER}",
          body: "Build completed successfully for $INSTITUTE_NAME"
    }
    failure {
      mail to: 'team@rushi.com',
          subject: "FAILURE: Build #${env.BUILD_NUMBER}",
          body: "Build failed for $INSTITUTE_NAME"
    }
  }
}
